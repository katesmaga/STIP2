    const mongoose = require("mongoose");

    const announcementSchema = new mongoose.Schema({
      title: {
        type: String,
        required: true,
      },
      description: {
        type: String,
        required: true,
      },
      price: {
        type: Number,
        required: true,
      },
    });
    
    const Announcement = mongoose.model("Announcement", announcementSchema);
    
    module.exports = Announcement;
